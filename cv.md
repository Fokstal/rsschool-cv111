```JavaScript
function toCamelCase(str){
  
  let camelCase = "";
  let bufferValue = 0;
  
  for (let i = 0; i < str.length; i++)
  {
    if (str[i] == "_" || str[i] == "-")
    {
      bufferValue = 1;
      continue
    }
    else if (bufferValue == 1)
    {
      bufferValue = 0;
      camelCase += str[i].toUpperCase();
    }
    else
    {
      camelCase += str[i];
    }
  }
  
  return camelCase;
}
```
